MIN_CLUSTER_SIZE = 400


def load_embeddings(df, column_name):
    return np.stack(df[column_name].values)


def perform_clustering(embeddings, min_samples, alpha, min_cluster_size=MIN_CLUSTER_SIZE):
    umap_model = UMAP(n_components=50, random_state=42)
    umap_embeddings = umap_model.fit_transform(StandardScaler().fit_transform(embeddings))

    hdbscan_model = HDBSCAN(min_cluster_size=min_cluster_size, 
                            min_samples=min_samples,
                            alpha=alpha,
                            cluster_selection_method='eom',
                            approx_min_span_tree=True
                           )
    labels = hdbscan_model.fit_predict(umap_embeddings)

    return labels, umap_embeddings


def tune_clustering(df, min_cluster_size=MIN_CLUSTER_SIZE, min_samples_range=range(5, 101, 5),
                    alpha_range=np.linspace(0.1, 2.5, 8)):
    df = df#.sample(8000, random_state=42)
    embeddings = load_embeddings(df, EMBEDDING_COLUMN)

    # Reduce dimensionality with progress bar
    print("Reducing dimensionality with UMAP...")
    reducer = umap.UMAP(n_components=50, random_state=42, verbose=True)
    reduced_embeddings = reducer.fit_transform(embeddings)

    def process_params(min_samples, alpha):
        hdbscan = HDBSCAN(min_cluster_size=min_cluster_size, min_samples=min_samples, alpha=alpha,
                          cluster_selection_method='eom', approx_min_span_tree=True)
        labels = hdbscan.fit_predict(reduced_embeddings)

        if len(set(labels)) > 1:
            silhouette = silhouette_score(reduced_embeddings, labels)
            n_clusters = len(set(labels)) - (1 if -1 in labels else 0)
            noise_proportion = np.sum(labels == -1) / len(labels)
            return {'min_samples': min_samples, 'alpha': alpha, 'silhouette_score': silhouette,
                    'n_clusters': n_clusters, 'noise_proportion': noise_proportion}
        return None

    print("Starting tuning process...")
    results = []
    total_iterations = len(min_samples_range) * len(alpha_range)
    
    with tqdm_notebook(total=total_iterations, desc="Tuning Progress") as pbar:
        for min_samples in min_samples_range:
            for alpha in alpha_range:
                result = process_params(min_samples, alpha)
                if result is not None:
                    results.append(result)
                pbar.update(1)

    results_df = pd.DataFrame(results)

    print("\nTuning completed. Displaying results:")
    display(results_df)

    best_params = results_df.loc[results_df['silhouette_score'].idxmax()]
    return best_params, results_df





def visualize_clusters(embeddings, labels, umap_embeddings):
    umap_2d = UMAP(n_components=2, random_state=42)
    umap_results_2d = umap_2d.fit_transform(embeddings)

    s_score = silhouette_score(umap_embeddings, labels)

    plt.figure(figsize=(10, 8))
    scatter = plt.scatter(umap_results_2d[:, 0],
                          umap_results_2d[:, 1],
                          alpha=0.75,
                          s=1,
                          c=labels,
                          cmap='Dark2')

    plt.colorbar(scatter, label='Cluster')
    plt.title(f'UMAP projection with HDBSCAN clustering n={len(labels.unique())}, S={s_score:.4f}')
    plt.xlabel('UMAP feature 1')
    plt.ylabel('UMAP feature 2')
    plt.tight_layout()
    plt.show()

import optuna
import numpy as np
import umap
import hdbscan
from sklearn.metrics import silhouette_score
import warnings
from multiprocessing import Pool, cpu_count
import contextlib

def objective(trial, embeddings):
    # UMAP parameters
    n_neighbors = trial.suggest_int('n_neighbors', 5, 50)
    min_dist = trial.suggest_float('min_dist', 0.0, 0.99)
    n_components = trial.suggest_int('n_components', 2, 100)

    # HDBSCAN parameters
    min_cluster_size = trial.suggest_int('min_cluster_size', 5, 100)
    min_samples = trial.suggest_int('min_samples', 1, 20)

    with warnings.catch_warnings():
        warnings.simplefilter("ignore")

        reducer = umap.UMAP(
            n_neighbors=n_neighbors,
            min_dist=min_dist,
            n_components=n_components,
            random_state=42
        )
        reduced_embeddings = reducer.fit_transform(embeddings)

        clusterer = hdbscan.HDBSCAN(
            min_cluster_size=min_cluster_size,
            min_samples=min_samples,
            metric='euclidean',
            cluster_selection_method='eom'
        )
        clusterer.fit(reduced_embeddings)

    labels = clusterer.labels_
    n_clusters = len(set(labels)) - (1 if -1 in labels else 0)
    noise_proportion = np.sum(labels == -1) / len(labels)

    if n_clusters > 1:
        sil_score = silhouette_score(reduced_embeddings, labels)
    else:
        sil_score = -1

    score = sil_score * (1 - noise_proportion) * n_clusters / 10
    return score

def optimize_clustering(embeddings, n_trials=100, n_jobs=None):
    n_jobs = n_jobs or cpu_count()
    
    def objective_wrapper(trial):
        return objective(trial, embeddings)

    study = optuna.create_study(direction='maximize')
    
    completed_trials = 0
    with Pool(processes=n_jobs) as pool:
        futures = []
        for _ in range(n_trials):
            trial = study.ask()
            future = pool.apply_async(objective_wrapper, (trial,))
            futures.append((trial, future))
        
        for trial, future in futures:
            try:
                result = future.get(timeout=300)  # 5 minutes timeout
                study.tell(trial, result)
                completed_trials += 1
                print(f"Completed trial {completed_trials}/{n_trials}")
            except Exception as e:
                print(f"Trial failed: {str(e)}")
                study.tell(trial, state=optuna.trial.TrialState.FAIL)

    print(f"\nOptimization completed. {completed_trials}/{n_trials} trials succeeded.")

    if completed_trials > 0:
        print("\nBest trial:")
        trial = study.best_trial
        print("  Value: ", trial.value)
        print("  Params: ")
        for key, value in trial.params.items():
            print(f"    {key}: {value}")
        return study.best_params, study.trials_dataframe()
    else:
        print("No trials completed successfully. Please check your objective function and parameters.")
        return None, study.trials_dataframe()
